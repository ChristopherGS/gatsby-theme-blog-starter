import { graphql } from 'gatsby';

import Layout from '../components/layout/layout';
import Seo from '../components/seo/Seo';
import SinglePostLayout from '../components/layout/SinglePostLayout';

<Layout>
      <Seo 
        title={props.data.mdx.frontmatter.title}
        description={props.data.mdx.frontmatter.description}
        tags={props.data.mdx.frontmatter.tags}
        image={props.data.mdx.frontmatter.image.childImageSharp.fluid}
        isBlogPost={true}
        slug={props.data.mdx.frontmatter.slug}
        date={props.data.mdx.frontmatter.date}
        update_date={props.data.mdx.frontmatter.update_date} />
      <SinglePostLayout data={props.data} />
</Layout>

export const postQuery = graphql`
  query blogPostBySlug($slug: String!) {
    mdx(frontmatter: { slug: {eq: $slug}}) {
      id
      body
      frontmatter{
        title
        description
        pagetitle
        summary
        date(formatString: "MMM D, YYYY")
        update_date(formatString: "MMM D, YYYY")
        tags
        slug
        image {
          childImageSharp {
            fluid(maxWidth: 700) {
              ...GatsbyImageSharpFluid
            }
          }
        }
      }
    }
  }
`
